-- ============================================================
-- UTILITY
-- ============================================================
local function new(class, props, parent)
	local inst = Instance.new(class)
	for k, v in pairs(props or {}) do inst[k] = v end
	if parent then inst.Parent = parent end
	return inst
end

local function corner(parent, radius)
	local c = Instance.new("UICorner")
	c.CornerRadius = UDim.new(0, radius or 6)
	c.Parent = parent
	return c
end

local function stroke(parent, color, thickness)
	local s = Instance.new("UIStroke")
	s.Color = color or Color3.fromRGB(32,32,32)
	s.Thickness = thickness or 1
	s.Parent = parent
	return s
end

local function tween(inst, props, t, style, dir)
	TweenService:Create(inst,
		TweenInfo.new(t or 0.15, style or Enum.EasingStyle.Quart, dir or Enum.EasingDirection.Out),
		props
	):Play()
end

-- Touch-aware drag helper — works for both mouse and touch
-- Returns a connection cleanup function
local function makeDraggable(handle, target, extraTarget)
	local dragging = false
	local dragStart, startPos

	local function beginDrag(pos)
		dragging  = true
		dragStart = pos
		startPos  = target.Position
	end
	local function moveDrag(pos)
		if not dragging then return end
		local d = pos - dragStart
		local np = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X,
		                     startPos.Y.Scale, startPos.Y.Offset + d.Y)
		target.Position = np
		if extraTarget then extraTarget.Position = np end
	end
	local function endDrag()
		dragging = false
	end

	handle.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or
		   i.UserInputType == Enum.UserInputType.Touch then
			beginDrag(Vector2.new(i.Position.X, i.Position.Y))
		end
	end)
	UserInputService.InputChanged:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseMovement or
		   i.UserInputType == Enum.UserInputType.Touch then
			moveDrag(Vector2.new(i.Position.X, i.Position.Y))
		end
	end)
	UserInputService.InputEnded:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 or
		   i.UserInputType == Enum.UserInputType.Touch then
			endDrag()
		end
	end)
end

-- Touch scroll helper — adds touch-drag scrolling to a ScrollingFrame
local function addTouchScroll(sf)
	local touching    = false
	local touchStart  = 0
	local canvasStart = 0

	sf.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.Touch then
			touching    = true
			touchStart  = i.Position.Y
			canvasStart = sf.CanvasPosition.Y
		end
	end)
	sf.InputChanged:Connect(function(i)
		if touching and i.UserInputType == Enum.UserInputType.Touch then
			local delta   = touchStart - i.Position.Y
			local maxY    = math.max(0, sf.AbsoluteCanvasSize.Y - sf.AbsoluteSize.Y)
			local newY    = math.clamp(canvasStart + delta, 0, maxY)
			sf.CanvasPosition = Vector2.new(0, newY)
		end
	end)
	sf.InputEnded:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.Touch then
			touching = false
		end
	end)
end

			-- On mobile use portrait width as reference, give a little extra room

	-- Smooth mouse-wheel scroll
	local scrollTargets = {}
	local function hookSlowScroll(frame)
		scrollTargets[frame] = 0
		frame.InputChanged:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.MouseWheel then
				local maxY = math.max(0, frame.AbsoluteCanvasSize.Y - frame.AbsoluteSize.Y)
				local target = math.clamp(scrollTargets[frame] + (-input.Position.Z * 50), 0, maxY)
				scrollTargets[frame] = target
				tween(frame, {CanvasPosition = Vector2.new(0, target)}, 0.25)
			end
		end)
		frame:GetPropertyChangedSignal("CanvasPosition"):Connect(function()
			if scrollTargets[frame] then
				local cy = frame.CanvasPosition.Y
				if math.abs(cy - scrollTargets[frame]) > 2 then
					scrollTargets[frame] = cy
				end
			end
		end)
	end
	hookSlowScroll(pageScroll)
	hookSlowScroll(sbScroll)

			end

						local p = (val - minV) / (maxV - minV)

						end
					end)

						local p = (val-minV)/(maxV-minV)

						if opts.Callback then opts.Callback(val) end

					return elem

						svCursor.Position  = UDim2.new(sat,0,1-val2,0)
